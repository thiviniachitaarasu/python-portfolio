# Sample inventory data analysis
import pandas as pd

# Create sample inventory data
inventory_data = {
    'Product': ['Item A', 'Item B', 'Item C', 'Item D', 'Item E'],
    'Stock': [150, 85, 200, 45, 120],

ax1.grid(axis='y', alpha=0.3)

class BusinessMetrics:
    """A simple calculator for common business metrics"""
    
    def __init__(self, revenue, costs, units_sold):
        self.revenue = revenue
        self.costs = costs
        self.units_sold = units_sold
    
    def profit_margin(self):
        """Calculate profit margin percentage"""
        profit = self.revenue - self.costs
        margin = (profit / self.revenue) * 100
        return round(margin, 2)
    
    def roi(self, investment):
        """Calculate Return on Investment"""
        profit = self.revenue - self.costs
        roi = ((profit - investment) / investment) * 100
        return round(roi, 2)
    
    def unit_economics(self):
        """Calculate per-unit metrics"""
        revenue_per_unit = self.revenue / self.units_sold
        cost_per_unit = self.costs / self.units_sold
        profit_per_unit = revenue_per_unit - cost_per_unit
        return {
            'Revenue per Unit': round(revenue_per_unit, 2),
            'Cost per Unit': round(cost_per_unit, 2),
            'Profit per Unit': round(profit_per_unit, 2)
        }
    
    def display_report(self):
        """Display comprehensive business report"""
        print("=" * 50)
        print("BUSINESS PERFORMANCE REPORT".center(50))
        print("=" * 50)
        print(f"\n Total Revenue: RM {self.revenue:,.2f}")
        print(f" Total Costs: RM {self.costs:,.2f}")
        print(f" Units Sold: {self.units_sold:,}")
        print(f"\n Profit Margin: {self.profit_margin()}%")
        print(f"\n Unit Economics:")
        for key, value in self.unit_economics().items():
            print(f"   {key}: RM {value}")
        print("=" * 50)

# Example usage
business = BusinessMetrics(revenue=50000, costs=32000, units_sold=500)
business.display_report()

class TaskManager:
    """A basic task management system"""
    
    def __init__(self):
        self.tasks = []
        self.task_id = 1
    
    def add_task(self, title, priority="Medium"):
        """Add a new task"""
        task = {
            'id': self.task_id,
            'title': title,
            'priority': priority,
            'status': 'Pending'
        }
        self.tasks.append(task)
        self.task_id += 1
        print(f" Task added: {title}")
    
    def complete_task(self, task_id):
        """Mark a task as complete"""
        for task in self.tasks:
            if task['id'] == task_id:
                task['status'] = 'Completed'
                print(f" Task {task_id} completed: {task['title']}")
                return
        print(f" Task {task_id} not found")
    
    def view_tasks(self):
        """Display all tasks"""
        if not self.tasks:
            print("No tasks available")
            return
        
        df_tasks = pd.DataFrame(self.tasks)
        print("\n" + "=" * 60)
        print("TASK LIST".center(60))
        print("=" * 60)
        print(df_tasks.to_string(index=False))
        
        # Task summary
        pending = len(df_tasks[df_tasks['status'] == 'Pending'])
        completed = len(df_tasks[df_tasks['status'] == 'Completed'])
        print("\n" + "=" * 60)
        print(f" Total Tasks: {len(self.tasks)} | Pending: {pending} | Completed: {completed}")

# Demo
tm = TaskManager()
tm.add_task("Prepare monthly report", "High")
tm.add_task("Update inventory database", "High")
tm.add_task("Follow up with suppliers", "Medium")
tm.add_task("Team meeting preparation", "Low")
tm.complete_task(1)
tm.complete_task(3)
tm.view_tasks()
